class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> pos;
        for (int i = 0; i < nums.size(); ++i) {
            int complement = target - nums[i];
            if (pos.find(complement) != pos.end()) {
                return {i, pos[complement]};
            }
            pos[nums[i]] = i;
        }
        
        return {};
    }
};
